 <script type="text/javascript>
  var cisla = new Array();

  for (var i = 0; i < 5; i++) {
   cisla[i] = parseInt(prompt('Zadaj číslo z klávesnice: '));
   document.write(cisla[i] + ' ');
  }

  var maximum = cisla[0];
  for (var i = 1; i < cisla.length; i++) {
   if (cisla[i] > maximum) {
    maximum = cisla[i];
   }
  }

  document.write("Najväčšie číslo je: " + maximum);
 </script>
